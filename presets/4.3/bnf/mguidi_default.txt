feature -SOFTSERIAL

#$ OPTION_GROUP BEGIN: DSHOT

#$ OPTION END

set dshot_bidir = ON

#$ OPTION END

#$ OPTION_GROUP END


#$ OPTION_GROUP BEGIN: Receiver

#$ OPTION BEGIN: Crossfire 50hz

feature RX_SERIAL

# Crossfire locked 50Hz

#$ OPTION END
